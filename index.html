-- ============================================================ -- VIEWS: para queries frequentes com performance -- ============================================================ -- View: resumo mensal por usuário CREATE OR REPLACE VIEW public.v_monthly_summary AS SELECT t.user_id, date_trunc('month', t.date) AS month, SUM(CASE WHEN t.type = 'income' THEN t.amount ELSE 0 END) AS total_income, SUM(CASE WHEN t.type = 'expense' THEN ABS(t.amount) ELSE 0 END) AS total_expenses, SUM(CASE WHEN t.type = 'expense' THEN ABS(t.amount) ELSE 0 END) / NULLIF(SUM(CASE WHEN t.type = 'income' THEN t.amount ELSE 0 END), 0) * 100 AS expense_ratio, COUNT(*) AS transaction_count FROM public.transactions t WHERE t.status != 'cancelled' GROUP BY t.user_id, date_trunc('month', t.date); -- View: gastos por categoria (30 dias) CREATE OR REPLACE VIEW public.v_category_expenses_30d AS SELECT t.user_id, c.id AS category_id, c.name AS category_name, c.icon, c.color, SUM(ABS(t.amount)) AS total, COUNT(*) AS count FROM public.transactions t JOIN public.categories c ON c.id = t.category_id WHERE t.type = 'expense' AND t.status != 'cancelled' AND t.date >= CURRENT_DATE - INTERVAL '30 days' GROUP BY t.user_id, c.id, c.name, c.icon, c.color; -- View: top estabelecimentos (30 dias) CREATE OR REPLACE VIEW public.v_top_merchants_30d AS SELECT t.user_id, t.merchant_name, SUM(ABS(t.amount)) AS total, COUNT(*) AS count, MAX(t.date) AS last_transaction FROM public.transactions t WHERE t.type = 'expense' AND t.merchant_name IS NOT NULL AND t.status != 'cancelled' AND t.date >= CURRENT_DATE - INTERVAL '30 days' GROUP BY t.user_id, t.merchant_name ORDER BY total DESC; -- ============================================================ -- FUNÇÕES: lógica de negócio no banco -- ============================================================ -- Função: calcular score financeiro CREATE OR REPLACE FUNCTION public.calculate_user_score(p_user_id UUID, p_month DATE DEFAULT date_trunc('month', CURRENT_DATE)) RETURNS TABLE( total_score INTEGER, spending_control INTEGER, budget_adherence INTEGER, income_consistency INTEGER, goals_progress INTEGER, subscriptions_ratio INTEGER, logging_habit INTEGER, emergency_fund INTEGER ) AS $$ DECLARE v_income DECIMAL; v_expenses DECIMAL; v_budget_e DECIMAL; v_budget_w DECIMAL; v_budget_i DECIMAL; v_tx_count INTEGER; v_goals_on_track INTEGER; v_goals_total INTEGER; v_sub_ratio DECIMAL; v_s1 INTEGER := 0; v_s2 INTEGER := 0; v_s3 INTEGER := 0; v_s4 INTEGER := 0; v_s5 INTEGER := 0; v_s6 INTEGER := 0; v_s7 INTEGER := 0; BEGIN -- Buscar perfil SELECT monthly_income, budget_essentials, budget_wants, budget_investments INTO v_income, v_budget_e, v_budget_w, v_budget_i FROM public.profiles WHERE id = p_user_id; -- Gastos do mês SELECT COALESCE(SUM(ABS(amount)), 0) INTO v_expenses FROM public.transactions WHERE user_id = p_user_id AND type = 'expense' AND date_trunc('month', date) = p_month AND status != 'cancelled'; -- s1: Controle de gastos (0-200) -- Ideal: gastar entre 80-95% do orçamento (sobrar algo sem desperdiçar) IF v_income > 0 THEN DECLARE ratio DECIMAL := v_expenses / v_income; BEGIN IF ratio <= 0.95 AND ratio >= 0.70 THEN v_s1 := 200; ELSIF ratio < 0.70 THEN v_s1 := ROUND(ratio / 0.70 * 160)::INTEGER; ELSIF ratio <= 1.05 THEN v_s1 := ROUND(200 - (ratio - 0.95) * 1000)::INTEGER; ELSE v_s1 := GREATEST(0, ROUND(200 - (ratio - 1) * 500)::INTEGER); END IF; END; END IF; -- s2: Aderência ao orçamento (0-200) — simplificado v_s2 := LEAST(200, ROUND(v_s1 * 0.9)::INTEGER); -- s3: Consistência de renda (0-150) — por ora fixo v_s3 := 120; -- s4: Progresso em metas (0-150) SELECT COUNT(*) INTO v_goals_total FROM public.goals WHERE user_id = p_user_id AND is_active = TRUE AND NOT is_completed; IF v_goals_total > 0 THEN SELECT COUNT(*) INTO v_goals_on_track FROM public.goals g WHERE g.user_id = p_user_id AND g.is_active = TRUE AND NOT g.is_completed AND (g.deadline IS NULL OR g.current_amount / NULLIF(g.target_amount, 0) >= (CURRENT_DATE - g.created_at::DATE) / NULLIF(g.deadline - g.created_at::DATE, 0)); v_s4 := ROUND(v_goals_on_track::DECIMAL / v_goals_total * 150)::INTEGER; END IF; -- s5: Controle de assinaturas (0-100) SELECT COALESCE(SUM(ABS(amount)), 0) INTO v_sub_ratio FROM public.transactions t JOIN public.categories c ON c.id = t.category_id AND c.slug = 'assinaturas' WHERE t.user_id = p_user_id AND date_trunc('month', t.date) = p_month; IF v_income > 0 THEN DECLARE sub_pct DECIMAL := v_sub_ratio / v_income; BEGIN IF sub_pct <= 0.10 THEN v_s5 := 100; ELSIF sub_pct <= 0.15 THEN v_s5 := 75; ELSIF sub_pct <= 0.20 THEN v_s5 := 50; ELSE v_s5 := GREATEST(0, ROUND(100 - sub_pct * 300)::INTEGER); END IF; END; END IF; -- s6: Hábito de registro (0-100) SELECT COUNT(DISTINCT date) INTO v_tx_count FROM public.transactions WHERE user_id = p_user_id AND source = 'manual' AND date >= CURRENT_DATE - 30; v_s6 := LEAST(100, ROUND(v_tx_count * 4.0)::INTEGER); -- 25 dias = 100 pts -- s7: Reserva de emergência (0-100) SELECT COALESCE(SUM(a.balance), 0) INTO v_sub_ratio FROM public.accounts a WHERE a.user_id = p_user_id AND a.type IN ('savings','investment'); IF v_income > 0 THEN v_s7 := LEAST(100, ROUND(v_sub_ratio / (v_income * 6) * 100)::INTEGER); END IF; RETURN QUERY SELECT (v_s1 + v_s2 + v_s3 + v_s4 + v_s5 + v_s6 + v_s7), v_s1, v_s2, v_s3, v_s4, v_s5, v_s6, v_s7; END; $$ LANGUAGE plpgsql SECURITY DEFINER; -- Função: projeção de gastos do fim do mês CREATE OR REPLACE FUNCTION public.project_month_end(p_user_id UUID) RETURNS TABLE( daily_avg DECIMAL, projected_total DECIMAL, projected_savings DECIMAL, days_remaining INTEGER ) AS $$ DECLARE v_income DECIMAL; v_expenses_mtd DECIMAL; v_day INTEGER := EXTRACT(DAY FROM CURRENT_DATE)::INTEGER; v_days_month INTEGER := EXTRACT(DAY FROM date_trunc('month', CURRENT_DATE) + INTERVAL '1 month' - INTERVAL '1 day')::INTEGER; BEGIN SELECT monthly_income INTO v_income FROM public.profiles WHERE id = p_user_id; SELECT COALESCE(SUM(ABS(amount)), 0) INTO v_expenses_mtd FROM public.transactions WHERE user_id = p_user_id AND type = 'expense' AND status != 'cancelled' AND date_trunc('month', date) = date_trunc('month', CURRENT_DATE); RETURN QUERY SELECT v_expenses_mtd / v_day, (v_expenses_mtd / v_day) * v_days_month, v_income - (v_expenses_mtd / v_day) * v_days_month, v_days_month - v_day; END; $$ LANGUAGE plpgsql SECURITY DEFINER;
